ODL Installation - Contrller
============================

Object=("server" "subnet" "router" "router-port" "port" "router-interface" "network")
for Obj in "${Object[@]}"; do
    for HashID in `openstack $Obj list -c ID | head -n -1 | tail -n +4 | cut -d ' ' -f 2`; do
        openstack $Obj delete $HashID
    done
done

sudo service neutron-server stop
sudo apt-get purge neutron-plugin-openvswitch-agent
sudo service openvswitch-switch stop

sudo rm -rf /var/log/openvswitch/*
rm -rf /etc/openvswitch/conf.db

sudo service openvswitch-switch start
ovs-vsctl show #This command must return the empty set except OpenVswitch

sudo pip install networking-odl
pip install networking_odl

sudo vim /etc/neutron/neutron.conf
	[DEFAULT]
	core_plugin = neutron.plugins.ml2.plugin.Ml2Plugin
	service_plugins = odl-router

    [qos]
    notification_drivers = odl-qos

sudo vim /etc/neutron/plugins/ml2/ml2_conf.ini
	[ml2]
	mechanism_drivers = opendaylight_v2
	type_drivers = local,flat,vlan,vxlan
	tenant_network_types = vxlan
	extension_drivers = port_security, qos

	[ml2_type_vxlan]
	vni_ranges = 1:1000

	[ml2_type_vlan]
	network_vlan_ranges = PHYSICAL_NETWORK:MIN_VLAN_ID:MAX_VLAN_ID
	network_vlan_ranges = physnet1,physnet2:1001:2000

	[securitygroup]
	enable_security_group = true

	[ml2_odl]
	username = <ODL_USERNAME>
	password = <ODL_PASSWORD>
	url = http://<ODL_IP_ADDRESS>:<ODL_PORT>/controller/nb/v2/neutron
	port_binding_controller = pseudo-agentdb-binding
	enable_dhcp_service = True

Compute/network
===============

sudo apt install openvswitch
sudo systemctl start openvswitch
/usr/share/openvswitch/scripts/ovs-ctl start

ovs-vsctl show 
#Open vSwitch ID

ovs-vsctl set Open_vSwitch <OPENVSWITCH ID> other_config={local_ip=<TUNNEL INTERFACE IP>}
ovs-vsctl set-manager tcp:<OPENDAYLIGHT MANAGEMENT IP>:6640

ovs-vsctl set-manager tcp:${ODL_IP_ADDRESS}:6640

ovs-vsctl add-br br-ex
ovs-vsctl add-port br-ex <INTERFACE NAME OF EXTERNAL NETWORK> #commonly eth0 or p2p1
ovs-vsctl show

sudo neutron-odl-ovs-hostconfig
ovs-vsctl show
