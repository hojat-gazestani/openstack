User Configuration
------------------
sudo useradd -d /home/ceph -m ceph
echo 'ceph:openstack' | sudo chpasswd
# OR
useradd ceph; echo openstack | passwd --stdin ceph				

echo "ceph ALL = (root) NOPASSWD: ALL" > /etc/sudoers.d/ceph	
chmod 0400 /etc/sudoers.d/ceph
su - ceph; sudo ls -a /root should list files
